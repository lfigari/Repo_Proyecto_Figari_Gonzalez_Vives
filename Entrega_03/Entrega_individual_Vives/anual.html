import pandas as pd
import altair as alt

# Cargar datos desde CSV
df = pd.read_csv("/content/pistacho grafico anual.csv", sep=";")

# Filtrar solo el producto 'pistacho'
df_pistacho = df[df['producto'].str.lower() == 'pistacho'].copy()

# Limpiar y convertir la columna 'Valor CIF acomulado' a n√∫mero
df_pistacho['Valor CIF acomulado'] = df_pistacho['Valor CIF acomulado'] \
    .str.replace('.', '', regex=False) \
    .str.replace(',', '.', regex=False) \
    .astype(float)

# Ordenar por a√±o
df_pistacho = df_pistacho.sort_values('a√±o')

# Crear columna de color seg√∫n el a√±o (2025 naranja, otros verde)
df_pistacho['color'] = df_pistacho['a√±o'].apply(lambda x: 'orange' if x == 2025 else 'green')

# Crear gr√°fico de barras con Altair
chart = alt.Chart(df_pistacho).mark_bar().encode(
    x=alt.X('a√±o:O', title='A√±o', sort='ascending'),
    y=alt.Y('Valor CIF acomulado:Q', title='Valor CIF acumulado'),
    color=alt.Color('color:N', scale=None, legend=None)
).properties(
    width=600,
    height=400,
    title='Valor CIF acumulado del Pistacho por a√±o (hasta diciembre, excepto 2025 que llega hasta septiembre)'
)

# A√±adir etiquetas de valor encima de cada barra
text = chart.mark_text(
    align='center',
    baseline='bottom',
    dy=-5  # Desplazamiento vertical del texto
).encode(
    text=alt.Text('Valor CIF acomulado:Q', format=',.0f')
)

# Mostrar gr√°fico
chart + text
